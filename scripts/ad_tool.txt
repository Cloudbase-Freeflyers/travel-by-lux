function getSetting(key, defaultValue) {
  const val = PropertiesService.getScriptProperties().getProperty(key);
  return val === null ? defaultValue : val === "true";
}

function setSetting(key, value) {
  PropertiesService.getScriptProperties().setProperty(key, value);
}

function getTextSetting_(key, defaultValue) {
  const val = PropertiesService.getScriptProperties().getProperty(key);
  return val === null || val === "" ? defaultValue : String(val);
}

function getNumberSetting_(key, defaultValue, min, max) {
  const raw = PropertiesService.getScriptProperties().getProperty(key);
  let n = Number(raw);
  if (!isFinite(n)) n = defaultValue;
  if (typeof min === "number" && n < min) n = min;
  if (typeof max === "number" && n > max) n = max;
  return n;
}

function getEnumSetting_(key, defaultValue, allowed) {
  const raw = String(getTextSetting_(key, defaultValue || "") || "").toLowerCase().trim();
  return allowed.indexOf(raw) >= 0 ? raw : defaultValue;
}

function htmlEscape_(text) {
  return String(text == null ? "" : text)
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#39;");
}

    "Function not available: " + fnName + "\n" +
    "Please make sure " + moduleFileHint + " is added and saved in this Apps Script project."
  );
}

function qaBatchConfigure() {
  const ui = SpreadsheetApp.getUi();
  const current = getTextSetting_("QA_BATCH_STEPS", "qaStandaloneFixAssets,qaStandaloneBrandComplianceAuto,qaLengthAuditRun,qaLengthFixAllTooLong");
  const resp = ui.prompt(
    "Configure QA Batch",
    "Enter comma-separated functions in run order.\nAllowed:\nqaStandaloneFixAssets, qaStandaloneBrandComplianceAuto, qaStandaloneBrandComplianceAudit, qaStandaloneBrandComplianceFix, qaLengthAuditRun, qaLengthFixAllTooLong, qaFixWeakAssets, generateBrandOptionBank, applyCampaignOptionPicks\n\nCurrent: " + current,
    ui.ButtonSet.OK_CANCEL
  );
  if (resp.getSelectedButton() !== ui.Button.OK) return;
  const next = (resp.getResponseText() || "").trim();
  setSetting("QA_BATCH_STEPS", next || current);
  ui.alert("QA batch steps saved.");
}

function qaBatchRunSelected() {
  const ui = SpreadsheetApp.getUi();
  const configured = getTextSetting_("QA_BATCH_STEPS", "qaStandaloneFixAssets,qaStandaloneBrandComplianceAuto,qaLengthAuditRun,qaLengthFixAllTooLong");
  const steps = configured.split(",").map(s => s.trim()).filter(Boolean);
  const brandQaMode = getEnumSetting_("BRAND_QA_MODE", DEFAULT_BRAND_QA_MODE, ["audit_only", "audit_and_fix"]);
  const allowed = {
    qaStandaloneFixAssets: menuQaStandaloneRunFix,
    qaStandaloneBrandComplianceAudit: menuQaStandaloneBrandAudit,
    qaStandaloneBrandComplianceFix: menuQaStandaloneBrandFix,
    qaLengthAuditRun: menuQaLengthAudit,
    qaLengthFixAllTooLong: menuQaLengthFixAllTooLong,
    qaFixWeakAssets: qaFixWeakAssets,
    generateBrandOptionBank: generateBrandOptionBank,
    applyCampaignOptionPicks: applyCampaignOptionPicks
  };
  let ran = 0;
  for (let i = 0; i < steps.length; i++) {
    const step = steps[i];
    if (step === "qaStandaloneBrandComplianceAuto") {
      menuQaStandaloneBrandAudit();
      ran++;
      if (brandQaMode === "audit_and_fix") {
        menuQaStandaloneBrandFix();
        ran++;
      }
      continue;
    }
    const fn = allowed[step];
    if (!fn) continue;
    fn();
    ran++;
  }
  ui.alert("QA batch complete. Steps run: " + ran + " | Brand QA mode: " + brandQaMode);
}

function renderAngleCheckboxesHtml_(options, selectedIds, groupType) {
  const selectedMap = {};
  (selectedIds || []).forEach(function(id) { selectedMap[String(id || "").toLowerCase()] = true; });
  const rows = (options || []).map(function(a) {
    const id = String(a && a.id || "").toLowerCase().trim();
    if (!id) return "";
    const label = String(a && a.name || id).trim();
    const checked = selectedMap[id] ? "checked" : "";
    const cls = groupType === "description" ? "angle-description" : "angle-headline";
    return '<label style="display:block;margin:4px 0;">' +
      '<input type="checkbox" class="' + cls + '" value="' + htmlEscape_(id) + '" ' + checked + ' onchange="syncSelectedAngles()">' +
      " " + htmlEscape_(label) + " (" + htmlEscape_(id) + ")" +
      "</label>";
  }).filter(Boolean);
  if (!rows.length) {
    return '<div style="color:#666;font-size:12px;">No angles found. Check your Brand Angles sheet and Angle library sheet setting.</div>';
  }
  return rows.join("");
}

function renderAssignmentRowsHtml_(options, assignmentMap, slotMax, rowCount, groupType) {
  const entries = Object.keys(assignmentMap || {}).map(function(id) {
    return { id: String(id || "").toLowerCase(), slot: Number(assignmentMap[id]) };
  }).filter(function(e) { return e.id && isFinite(e.slot); }).sort(function(a, b) { return a.slot - b.slot; });
  const angleChoices = ['<option value="">-- Select angle --</option>'].concat(
    (options || []).map(function(a) {
      const id = String(a && a.id || "").toLowerCase().trim();
      const name = String(a && a.name || id).trim();
      if (!id) return "";
      return '<option value="' + htmlEscape_(id) + '">' + htmlEscape_(name) + " (" + htmlEscape_(id) + ")" + "</option>";
    }).filter(Boolean)
  );
  const slotChoices = ['<option value="">-- Slot --</option>'];
  for (let s = 1; s <= slotMax; s++) slotChoices.push('<option value="' + s + '">' + s + "</option>");
  const rows = [];
  for (let i = 0; i < rowCount; i++) {
    const row = entries[i] || { id: "", slot: "" };
    const angleOptions = angleChoices.map(function(opt) {
      if (!row.id) return opt;
      const selectedNeedle = 'value="' + htmlEscape_(row.id) + '"';
      return opt.indexOf(selectedNeedle) >= 0 ? opt.replace(">", " selected>") : opt;
    }).join("");
    const slotOptions = slotChoices.map(function(opt) {
      if (!row.slot) return opt;
      const selectedNeedle = 'value="' + row.slot + '"';
      return opt.indexOf(selectedNeedle) >= 0 ? opt.replace(">", " selected>") : opt;
    }).join("");
    rows.push(
      '<div data-assign-group="' + htmlEscape_(groupType) + '" style="display:grid;grid-template-columns:1fr 90px;gap:8px;margin:6px 0;">' +
      '<select class="assign-angle" onchange="syncAssignmentInputs()" style="width:100%;box-sizing:border-box;">' + angleOptions + "</select>" +
      '<select class="assign-slot" onchange="syncAssignmentInputs()" style="width:100%;box-sizing:border-box;">' + slotOptions + "</select>" +
      "</div>"
    );
  }
  return rows.join("");
}

function extractPromptSettings(payload) {
  const websiteUrl = String(payload && payload.websiteUrl || "").trim();
  const docUrl = String(payload && payload.docUrl || "").trim();
  const sourceText = String(payload && payload.sourceText || "").trim();
  const apply = !!(payload && payload.apply);

  let docText = "";
  if (docUrl) {
    docText = getGoogleDocTextFromUrl_(docUrl);
  }

  const mergedSource = [
    websiteUrl ? ("Website: " + websiteUrl) : "",
    docText ? ("Google Doc Content:\n" + docText) : "",
    sourceText ? ("Pasted Content:\n" + sourceText) : ""
  ].filter(Boolean).join("\n\n");

  if (!mergedSource) {
    throw new Error("Provide at least one source: website URL, Google Doc URL, or pasted text.");
  }

  // #region agent log
  agentDebugLog_("H12", "ad_tool.txt:extractPromptSettings", "Extractor inputs received", {
    hasWebsiteUrl: !!websiteUrl,
    hasDocUrl: !!docUrl,
    hasSourceText: !!sourceText,
    apply: apply
  }, "pre-fix");
  // #endregion

  const extractionPrompt = buildExtractionPrompt_(mergedSource);
  const rawExtracted = callOpenAIJson(extractionPrompt, {
    model: getTextSetting_("OPENAI_MODEL", DEFAULT_OPENAI_MODEL),
    temperature: 0.2
  });
  const extracted = normalizeExtractedSettings_(rawExtracted);
  const manifestSummary = buildBrandManifestSummaryFromExtract_(extracted);

  if (apply) {
    applyExtractedSettings_(extracted);
  }

  return {
    message: apply ? "Extracted and applied to settings." : "Extracted successfully.",
    extracted: extracted,
    manifestSummary: manifestSummary
  };
}

function getGoogleDocTextFromUrl_(docUrl) {
  const match = String(docUrl).match(/\/d\/([a-zA-Z0-9_-]+)/);
  if (!match || !match[1]) throw new Error("Invalid Google Doc URL.");
  const docId = match[1];
  const doc = DocumentApp.openById(docId);
  return doc.getBody().getText().slice(0, 25000);
}

  };
  return {
    brandName: String(out.brandName || ""),
    brandWebsite: String(out.brandWebsite || ""),
    positioning1: String(out.positioning1 || ""),
    positioning2: String(out.positioning2 || ""),

  const maxRetries = getNumberSetting_("MAX_RETRIES", DEFAULT_MAX_RETRIES, 0, 5);
  const retryBackoffMs = getNumberSetting_("RETRY_BACKOFF_MS", DEFAULT_RETRY_BACKOFF_MS, 0, 10000);
  const retryTemperature = getNumberSetting_("RETRY_TEMPERATURE", DEFAULT_RETRY_TEMPERATURE, 0, 1.5);
  const promptCfg = getPromptConfig_();
  // #region agent log
  agentDebugLog_("H10", "ad_tool.txt:generateAds:config", "Generation runtime config", {
    lockShared: lockShared,
    lockSharedDescriptions: lockSharedDescriptions,
    specificHeadlineCount: specificHeadlineCount,
    specificDescriptionCount: specificDescriptionCount,
    rowLimit: rowLimit,
    batchDelayMs: batchDelayMs,
    model: model,
    temperature: temperature,
    maxRetries: maxRetries,
    retryBackoffMs: retryBackoffMs,
    retryTemperature: retryTemperature
  }, "pre-fix");
  // #endregion
  const fixedAssets = readFixedAssetsConfig_(ss, getTextSetting_("FIXED_ASSETS_SHEET_NAME", DEFAULT_FIXED_ASSETS_SHEET));
  let sharedHeadlines = lockShared ? getSharedHeadlinesBaseline_(adData, specificHeadlineCount) : null;
  let sharedDescriptions = lockSharedDescriptions ? getSharedDescriptionsBaseline_(adData, specificDescriptionCount) : null;
  if (fixedAssets.warning) {
    logSheet.appendRow([new Date(), "", "", "FIXED_ASSETS_WARN", fixedAssets.warning]);
  }

  let processed = 0;

  for (let i = 1; i < adData.length; i++) {

    if (processed >= rowLimit) break;

    const adGroup = (adData[i][1] == null ? "" : String(adData[i][1])).trim();
    if (!adGroup) {
      logSheet.appendRow([new Date(), i + 1, "", "SKIP_EMPTY", "Ad group is blank"]);
      continue;
    }

    // Only skip if already complete (15 headlines + 4 descriptions)
    const existingHeadlines = adData[i].slice(3, 18)
      .filter(v => v && v.toString().trim());
    const existingDescriptions = adData[i].slice(18, 22)
      .filter(v => v && v.toString().trim());

    if (existingHeadlines.length >= 15 && existingDescriptions.length >= 4) {
      logSheet.appendRow([new Date(), i + 1, adGroup, "SKIP_COMPLETE", "Row already has 15 headlines and 4 descriptions"]);
      continue;
    }

    const normalizedAdGroup = adGroup.toString().toLowerCase().trim();

    // Attempt 1: exact normalized match
    let keywordColumn = keywordHeaders.findIndex(h => {
      if (!h) return false;
      return h.toString().toLowerCase().trim() === normalizedAdGroup;
    });

    // Attempt 2: partial match (header contains ad group or vice versa)
    if (keywordColumn === -1) {
      keywordColumn = keywordHeaders.findIndex(h => {
        if (!h) return false;
        const hNorm = h.toString().toLowerCase().trim();
        return hNorm.includes(normalizedAdGroup) || normalizedAdGroup.includes(hNorm);
      });
    }

    let keywords = [];

    if (keywordColumn !== -1) {
      for (let r = 1; r < keywordData.length; r++) {
        const kw = keywordData[r][keywordColumn];
        if (kw && kw.toString().trim()) keywords.push(kw);
      }
    }

    // Fallback: no keywords found — generate from ad group name and log
    if (!keywords.length) {
      keywords = [adGroup.toString().trim()];
      logSheet.appendRow([
        new Date(),
        i + 1,
        adGroup,
        "NO_KEYWORDS",
        "Generated from ad group name only. Keyword column not found or empty."
      ]);
      Logger.log("Ad Gen Fallback (row " + (i + 1) + "): " + adGroup + " — no keywords, using ad group name");
    }

    let generated = false;
    let lastError = "";
    for (let attempt = 0; attempt <= maxRetries; attempt++) {
      const attemptNo = attempt + 1;
      const attemptTemp = attempt === 0 ? temperature : retryTemperature;
      try {
        const retryNote = attempt > 0
          ? "\n\nRETRY REQUIREMENT:\nReturn strict JSON with exactly 15 unique headlines and 4 descriptions."
          : "";
        const prompt = buildPrompt(adGroup, keywords) + retryNote;
        const aiResult = callOpenAIJson(prompt, {
          model: model,
          temperature: attemptTemp
        });

        if (!aiResult.headlines || aiResult.headlines.length !== 15) throw new Error("Bad headlines");
        if (!aiResult.descriptions || aiResult.descriptions.length !== 4) throw new Error("Bad descriptions");

        const finalHeadlines = aiResult.headlines.slice().map(v => String(v == null ? "" : v).trim());
        const finalDescriptions = aiResult.descriptions.slice().map(v => String(v == null ? "" : v).trim());

        if (lockShared) {
          if (!sharedHeadlines && hasCompleteSharedHeadlines_(finalHeadlines, specificHeadlineCount)) {
            sharedHeadlines = finalHeadlines.slice(specificHeadlineCount, 15);
          }
          if (sharedHeadlines && sharedHeadlines.length === (15 - specificHeadlineCount)) {
            for (let h = specificHeadlineCount; h < 15; h++) {
              finalHeadlines[h] = sharedHeadlines[h - specificHeadlineCount];
            }
          }
        }

        if (lockSharedDescriptions) {
          if (!sharedDescriptions && hasCompleteSharedDescriptions_(finalDescriptions, specificDescriptionCount)) {
            sharedDescriptions = finalDescriptions.slice(specificDescriptionCount, 4);
          }
          if (sharedDescriptions && sharedDescriptions.length === (4 - specificDescriptionCount)) {
            for (let d = specificDescriptionCount; d < 4; d++) {
              finalDescriptions[d] = sharedDescriptions[d - specificDescriptionCount];
            }
          }
        }

        applyAngleAssignmentsByPosition_(finalHeadlines, finalDescriptions, promptCfg, logSheet, i + 1, adGroup);
        applyFixedAssetsByPosition_(finalHeadlines, finalDescriptions, fixedAssets, logSheet, i + 1, adGroup);
        ensureUniqueHeadlines_(finalHeadlines, {
          adGroup: adGroup,
          keywords: keywords,
          logSheet: logSheet,
          rowNumber: i + 1,
          lockedPositions: Object.keys((fixedAssets && fixedAssets.headlineByPosition) || {}).map(function(k) { return Number(k); })
        });

        const output = [...finalHeadlines, ...finalDescriptions];
        adSheet.getRange(i + 1, 4, 1, output.length).setValues([output]);

        processed++;
        generated = true;
        logSheet.appendRow([new Date(), i + 1, adGroup, "GENERATED", "Assets generated successfully on attempt " + attemptNo]);
        Logger.log("Generated row: " + (i + 1) + " | " + adGroup + " | attempt " + attemptNo);
        Utilities.sleep(batchDelayMs);
        break;
      } catch (error) {
        lastError = String(error);
        const canRetry = isRecoverableGenerationError_(lastError) && attempt < maxRetries;
        if (canRetry) {
          logSheet.appendRow([new Date(), i + 1, adGroup, "RETRY_" + attemptNo, lastError]);
          Utilities.sleep(retryBackoffMs);
          continue;
        }
        logSheet.appendRow([new Date(), i + 1, adGroup, "ERROR", lastError + " | attempts=" + attemptNo]);
        Logger.log("Generation failed: " + adGroup + " | " + lastError);
        break;
      }
    }
    if (!generated) continue;
  }
}

function isRecoverableGenerationError_(errorText) {
  const text = String(errorText || "").toLowerCase();
  return (
    text.includes("bad headlines") ||
    text.includes("bad descriptions") ||
    text.includes("json") ||
    text.includes("unexpected token") ||
    text.includes("unterminated")
  );
}

function qaFixWeakAssets() {

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const adSheetName = getTextSetting_("AD_SHEET_NAME", "Ad Copy");
  const keywordSheetName = getTextSetting_("KEYWORD_SHEET_NAME", "List Of Keywords");
  const adSheet = ss.getSheetByName(adSheetName);
  const keywordSheet = ss.getSheetByName(keywordSheetName);
  if (!adSheet || !keywordSheet) {
    SpreadsheetApp.getUi().alert(
      "Missing sheet",
      "Check settings: ad sheet '" + adSheetName + "', keyword sheet '" + keywordSheetName + "'.",
      SpreadsheetApp.getUi().ButtonSet.OK
    );
    return;
  }
  const logSheet = getOrCreateQaLogSheet_();

  const adData = adSheet.getDataRange().getValues();
  const keywordData = keywordSheet.getDataRange().getValues();
  const keywordHeaders = keywordData[0];
  adSheet.getRange(1, 24).setValue("QA Score");

  const lockShared = getSetting("LOCK_SHARED_HEADLINES", false);
  const lockSharedDescriptions = getSetting("LOCK_SHARED_DESCRIPTIONS", false);
  const specificHeadlineCount = getNumberSetting_("SPECIFIC_HEADLINE_COUNT", DEFAULT_SPECIFIC_HEADLINE_COUNT, 3, 5);
  const specificDescriptionCount = getNumberSetting_("SPECIFIC_DESCRIPTION_COUNT", DEFAULT_SPECIFIC_DESCRIPTION_COUNT, 1, 3);
  let sharedHeadlines = lockShared ? getSharedHeadlinesBaseline_(adData, specificHeadlineCount) : null;
  let sharedDescriptions = lockSharedDescriptions ? getSharedDescriptionsBaseline_(adData, specificDescriptionCount) : null;

  for (let i = 1; i < adData.length; i++) {

    const rowNumber = i + 1;
    const adGroup = (adData[i][1] == null ? "" : String(adData[i][1])).trim();
    if (!adGroup) continue;
    // #region agent log
    agentDebugLog_("H7", "ad_tool.txt:qaFixWeakAssets:rowStart", "QA row processing started", {
      row: rowNumber,
      adGroup: String(adGroup),
      lockShared: lockShared,
      lockSharedDescriptions: lockSharedDescriptions
    }, "pre-fix");
    // #endregion

    const keywords = getKeywordsForAdGroup_(adGroup, keywordHeaders, keywordData);

    let headlines = adData[i].slice(3, 18).map(v => (v == null ? "" : String(v)).trim());
    let descriptions = adData[i].slice(18, 22).map(v => (v == null ? "" : String(v)).trim());
    const rowScores = [];

    // QA mode should repair existing assets only, not create new content in blank rows.
    if (headlines.concat(descriptions).every(v => !v)) continue;

    let allAssets = headlines.concat(descriptions);

    for (let h = 0; h < headlines.length; h++) {
      if (!headlines[h]) continue;
      const score = scoreAsset_(headlines[h], "headline", h, adGroup);
      rowScores.push(score);

      if (lockShared && h >= specificHeadlineCount) {
        if (!sharedHeadlines && hasCompleteSharedHeadlines_(headlines, specificHeadlineCount)) {
          sharedHeadlines = headlines.slice(specificHeadlineCount, 15);
        }
        if (sharedHeadlines && headlines[h] !== sharedHeadlines[h - specificHeadlineCount]) {
          adSheet.getRange(rowNumber, 4 + h).setValue(sharedHeadlines[h - specificHeadlineCount]).setBackground("#cfe2f3");
          headlines[h] = sharedHeadlines[h - specificHeadlineCount];
        } else {
          adSheet.getRange(rowNumber, 4 + h).setBackground(null);
        }
        continue;
      }

      const issues = analyzeSingleAsset_(headlines[h], "headline", h, adGroup);
      if (!issues.length) {
        adSheet.getRange(rowNumber, 4 + h).setBackground(null);
        continue;
      }

      const fixed = fixSingleAsset_(headlines[h], "headline", adGroup, keywords, headlines, descriptions);

      if (!fixed || allAssets.includes(fixed)) continue;

      adSheet.getRange(rowNumber, 4 + h).setValue(fixed).setBackground("#d9ead3");

      logQaChange_(logSheet, rowNumber, adGroup,
        columnToLetter_(4 + h),
        "Headline " + (h+1),
        headlines[h],
        fixed,
        issues.join(" | ")
      );
    }

    for (let d = 0; d < descriptions.length; d++) {
      if (!descriptions[d]) continue;
      rowScores.push(scoreAsset_(descriptions[d], "description", d, adGroup));

      if (lockSharedDescriptions && d >= specificDescriptionCount) {
        if (!sharedDescriptions && hasCompleteSharedDescriptions_(descriptions, specificDescriptionCount)) {
          sharedDescriptions = descriptions.slice(specificDescriptionCount, 4);
        }
        if (sharedDescriptions && descriptions[d] !== sharedDescriptions[d - specificDescriptionCount]) {
          adSheet.getRange(rowNumber, 19 + d).setValue(sharedDescriptions[d - specificDescriptionCount]).setBackground("#cfe2f3");
          descriptions[d] = sharedDescriptions[d - specificDescriptionCount];
          allAssets[15 + d] = descriptions[d];
        } else {
          adSheet.getRange(rowNumber, 19 + d).setBackground(null);
        }
        continue;
      }

      const issues = analyzeSingleAsset_(descriptions[d], "description", d, adGroup);
      if (!issues.length) {
        adSheet.getRange(rowNumber, 19 + d).setBackground(null);
        continue;
      }

      const fixed = fixSingleAsset_(descriptions[d], "description", adGroup, keywords, headlines, descriptions);

      if (!fixed || allAssets.includes(fixed)) continue;

      adSheet.getRange(rowNumber, 19 + d).setValue(fixed).setBackground("#d9ead3");
      logQaChange_(logSheet, rowNumber, adGroup,
        columnToLetter_(19 + d),
        "Description " + (d+1),
        descriptions[d],
        fixed,
        issues.join(" | ")
      );
      descriptions[d] = fixed;
      allAssets[15 + d] = fixed;
    }

    if (rowScores.length) {
      const avgScore = Math.round(rowScores.reduce((a, b) => a + b, 0) / rowScores.length);
      const scoreCell = adSheet.getRange(rowNumber, 24);
      scoreCell.setValue(avgScore);
      if (avgScore < 70) {
        scoreCell.setBackground("#f4cccc");
      } else if (avgScore < 85) {
        scoreCell.setBackground("#fff2cc");
      } else {
        scoreCell.setBackground("#d9ead3");
      }
    } else {
      adSheet.getRange(rowNumber, 24).setValue("").setBackground(null);
    }
  }

  SpreadsheetApp.getUi().alert("QA complete");
}

function analyzeSingleAsset_(text, type, index, adGroup) {
  if (!text) return ["empty"];

  const issues = [];
  const lower = String(text).toLowerCase();
  const cfg = getPromptConfig_();
  const banned = cfg.bannedWords || [];
  const generic = cfg.offShelfWords || [];
  const vague = ["explore", "discover", "experience", "travel better"];

  if (type === "headline" && text.length > 30) issues.push("too long");
  if (type === "description" && text.length > 90) issues.push("too long");
  if (type === "description" && text.length < 60) issues.push("too short");

  banned.forEach(w => {
    const term = String(w || "").toLowerCase().trim();
    if (term && lower.includes(term)) issues.push("AI wording");
  });

  generic.forEach(w => {
    const term = String(w || "").toLowerCase().trim();
    if (term && lower.includes(term)) issues.push("generic/off-the-shelf");
  });

  vague.forEach(w => {
    if (lower.includes(w)) issues.push("too vague");
  });

  if (type === "headline" && index < 3) {
    const words = String(adGroup || "").toLowerCase().split(/\s+/).filter(Boolean);
    const match = words.some(w => lower.includes(w));
    if (!match) issues.push("not ad group specific");
  }

  if (String(text).split(" ").length <= 2) issues.push("too vague");

  return [...new Set(issues)];
}

function fixSingleAsset_(text, type, adGroup, keywords, headlines, descriptions) {
  const useExample = getSetting("USE_EXAMPLE_STYLE", getSetting("USE_LIRAN_STYLE", false));
  const exampleReference = useExample ? getExampleStyleReference_() : "";
  // #region agent log
  agentDebugLog_("H8", "ad_tool.txt:fixSingleAsset_", "QA rewrite style context", {
    adGroup: String(adGroup || ""),
    type: type,
    useExample: useExample,
    hasExampleRef: !!exampleReference
  }, "pre-fix");
  // #endregion

  const cfg = getPromptConfig_();
  const prompt = `
Rewrite this ${type} for a luxury travel ad.

Original:
${text}

Ad group:
${adGroup}

Keywords:
${(keywords || []).join(", ")}

${useExample ? `
STYLE REFERENCE (VERY IMPORTANT):
${exampleReference}
Do NOT copy directly. Match tone and specificity.
` : ""}

Rules:
- Must clearly reflect private guided tours, custom itineraries, or exclusive access
- Avoid banned words: ${(cfg.bannedWords || []).join(", ")}
- Avoid generic/off-the-shelf phrasing: ${(cfg.offShelfWords || []).join(", ")}
- If the event/context is Gemini, use "celebrate", "commemorate", or "milestone"
- For Gemini context, shift tone to festive and joyous
- Prefer "Celebrate your Gemini milestone" over neutral booking phrasing
- Keep tone human, clear, and specific
- Do not duplicate existing lines
- ${type === "headline" ? "Max 30 characters including spaces" : "60 to 90 characters including spaces"}

Existing assets:
${headlines.concat(descriptions).join(" | ")}

Return ONLY the rewritten line.
`;
  try {
    return callOpenAIText(prompt).trim();
  } catch (e) {
    return null;
  }
}

function callOpenAIText(prompt, options) {
  const apiKey = getApiKey();
  const model = options && options.model ? String(options.model) : DEFAULT_OPENAI_MODEL;
  const temperature = options && options.temperature != null
    ? Number(options.temperature)
    : DEFAULT_OPENAI_TEMPERATURE;

  const response = UrlFetchApp.fetch(
    "https://api.openai.com/v1/chat/completions",
    {
      method: "post",
      contentType: "application/json",
      headers: { Authorization: "Bearer " + apiKey },
      payload: JSON.stringify({
        model: model,
        messages: [{ role: "user", content: prompt }],
        temperature: temperature
      })
    }
  );
  const json = JSON.parse(response.getContentText());
  const content = json && json.choices && json.choices[0] && json.choices[0].message
    ? json.choices[0].message.content
    : "";
  return String(content || "").replace(/```/g, "").trim();
}

function getExampleStyleReference_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const exampleSheetName = getTextSetting_("EXAMPLE_SHEET_NAME", "Ad copy example");
  const exampleSheet = ss.getSheetByName(exampleSheetName);
  if (!exampleSheet) return "";
  const values = exampleSheet.getDataRange().getDisplayValues();
  const lines = [];
  for (let r = 0; r < values.length; r++) {
    for (let c = 0; c < values[r].length; c++) {
      const cell = (values[r][c] || "").toString().trim();
      if (cell) lines.push(cell);
    }
  }
  return lines.join("\n").slice(0, 4000);
}

function parseCsvOrLines_(text, fallbackList) {
  const raw = String(text == null ? "" : text).trim();
  const src = raw || fallbackList.join(",");
  return src
    .split(/[\n,]/)
    .map(s => s.trim())
    .filter(Boolean);
}

function callOpenAIJson(prompt, options) {

  const apiKey = getApiKey();
  const model = options && options.model ? String(options.model) : DEFAULT_OPENAI_MODEL;
  const temperature = options && options.temperature != null
    ? Number(options.temperature)
    : DEFAULT_OPENAI_TEMPERATURE;

  const response = UrlFetchApp.fetch(
    "https://api.openai.com/v1/chat/completions",
    {
      method: "post",
      contentType: "application/json",
      headers: { Authorization: "Bearer " + apiKey },
      payload: JSON.stringify({
        model: model,
        response_format: { type: "json_object" },
        messages: [{ role: "user", content: prompt }],
        temperature: temperature
      })
    }
  ); // ✅ THIS WAS MISSING

  const json = JSON.parse(response.getContentText());

  let content = json.choices[0].message.content;

  content = content.replace(/```json/g,"").replace(/```/g,"").trim();

  return JSON.parse(content);
}

function readFixedAssetsConfig_(ss, sheetName) {
  const name = String(sheetName || DEFAULT_FIXED_ASSETS_SHEET);
  const sheet = ss.getSheetByName(name);
  if (!sheet) {
    return { headlineByPosition: {}, descriptionByPosition: {}, warning: "Fixed assets sheet not found: " + name };
  }
  const values = sheet.getDataRange().getValues();
  const headers = (values[0] || []).map(v => String(v == null ? "" : v).trim().toLowerCase());
  const idxEnabled = findHeaderIndex_(headers, ["enabled", "active", "apply"]);
  const idxType = findHeaderIndex_(headers, ["type", "asset type"]);
  const idxPosition = findHeaderIndex_(headers, ["position", "slot", "fixed position"]);
  const idxText = findHeaderIndex_(headers, ["text", "asset", "copy"]);
  if (idxType < 0 || idxPosition < 0 || idxText < 0) {
    return { headlineByPosition: {}, descriptionByPosition: {}, warning: "Fixed assets sheet missing required headers (type, position, text)." };
  }

  const headlineByPosition = {};
  const descriptionByPosition = {};
  for (let i = 1; i < values.length; i++) {
    const row = values[i];
    const enabled = idxEnabled < 0 ? true : toBoolCell_(row[idxEnabled]);
    if (!enabled) continue;
    const typeRaw = String(row[idxType] == null ? "" : row[idxType]).toLowerCase().trim();
    const text = String(row[idxText] == null ? "" : row[idxText]).trim();
    const position = Number(row[idxPosition]);
    if (!text || !isFinite(position)) continue;
    if (typeRaw === "headline" && position >= 1 && position <= 15 && text.length <= 30) {
      headlineByPosition[position] = text;
    } else if (typeRaw === "description" && position >= 1 && position <= 4 && text.length <= 90) {
      descriptionByPosition[position] = text;
    }
  }
  return { headlineByPosition: headlineByPosition, descriptionByPosition: descriptionByPosition, warning: "" };
}

}

function toBoolCell_(v) {
  if (v === true) return true;

  if (s === "0" || s === "false" || s === "no" || s === "n" || s === "off") return false;
  return true;
}

function parseAngleMinMap_(raw) {
  const out = {};
  const parts = String(raw == null ? "" : raw).split(/[\n,]/);
  for (let i = 0; i < parts.length; i++) {
    const token = String(parts[i] || "").trim();
    if (!token) continue;
    const m = token.match(/^([a-z0-9 _-]+)\s*:\s*(\d+)$/i);
    if (!m) continue;
    const key = String(m[1] || "").trim().toLowerCase();
    const count = Math.max(0, Math.min(4, Number(m[2])));
    if (!key) continue;
    if (!count) continue;
    out[key] = count;
  }
  return out;
}

function parseCsvIds_(raw, maxCount) {
  const seen = {};
  const out = [];
  const parts = String(raw == null ? "" : raw).split(/[\n,]/);
  for (let i = 0; i < parts.length; i++) {
    const id = String(parts[i] || "").toLowerCase().trim();
    if (!id || seen[id]) continue;
    seen[id] = true;
    out.push(id);
    if (typeof maxCount === "number" && out.length >= maxCount) break;
  }
  return out;
}

function parseAngleAssignmentMap_(raw, maxSlot) {
  const out = {};
  const entries = String(raw == null ? "" : raw).split(/[\n,]/);
  for (let i = 0; i < entries.length; i++) {
    const token = String(entries[i] || "").trim();
    if (!token) continue;
    const m = token.match(/^([a-z0-9 _-]+)\s*:\s*(\d+)$/i);
    if (!m) continue;
    const id = String(m[1] || "").toLowerCase().trim();
    const slot = Number(m[2]);
    if (!id || !isFinite(slot)) continue;
    if (slot < 1 || slot > maxSlot) continue;
    out[id] = slot;
  }
  return out;
}

function normalizeHeadlineKey_(text) {
  return String(text == null ? "" : text).toLowerCase().replace(/\s+/g, " ").trim();

function ensureUniqueHeadlines_(headlines, opts) {
  const used = {};
  const locked = {};
  (opts && opts.lockedPositions || []).forEach(function(p) { locked[Number(p)] = true; });
  for (let i = 0; i < headlines.length; i++) {
    const current = String(headlines[i] == null ? "" : headlines[i]).trim();
    if (!current) continue;
    const key = normalizeHeadlineKey_(current);
    if (!used[key]) {
      used[key] = i + 1;
      continue;
    }
    const pos = i + 1;
    if (locked[pos]) {
      if (opts && opts.logSheet) {
        opts.logSheet.appendRow([new Date(), opts.rowNumber || "", opts.adGroup || "", "DUPLICATE_LOCKED", "Duplicate locked headline at H" + pos + ": " + current]);
      }
      continue;
    }
    let replacement = buildDeterministicUniqueHeadline_(current, opts && opts.adGroup, used);
    if (!replacement) {
      replacement = rewriteDuplicateHeadlineAi_(current, opts && opts.adGroup, opts && opts.keywords, Object.keys(used));
    }
    replacement = String(replacement || "").trim();
    const repKey = normalizeHeadlineKey_(replacement);
    if (!replacement || used[repKey] || replacement.length > 30) {
      if (opts && opts.logSheet) {
        opts.logSheet.appendRow([new Date(), opts.rowNumber || "", opts.adGroup || "", "DUPLICATE_UNRESOLVED", "Could not dedupe headline at H" + pos + ": " + current]);
      }
      continue;
    }
    headlines[i] = replacement;
    used[repKey] = pos;
    if (opts && opts.logSheet) {
      opts.logSheet.appendRow([new Date(), opts.rowNumber || "", opts.adGroup || "", "DUPLICATE_FIXED", "H" + pos + ": " + current + " -> " + replacement]);
    }
  }
}

function buildDeterministicUniqueHeadline_(headline, adGroup, usedMap) {
  const base = String(headline == null ? "" : headline).trim();
  if (!base) return "";
  const tokens = String(adGroup == null ? "" : adGroup)
    .split(/\s+/)
    .map(function(v) { return String(v || "").trim(); })
    .filter(function(v) { return v.length > 2; });
  for (let i = 0; i < tokens.length; i++) {
    const cand = (base + " " + tokens[i]).replace(/\s+/g, " ").trim();
    if (cand.length > 30) continue;
    if (!usedMap[normalizeHeadlineKey_(cand)]) return cand;
  }
  const suffixes = ["Private", "Custom", "Luxury", "Expert"];
  for (let j = 0; j < suffixes.length; j++) {
    const cand = (base + " " + suffixes[j]).replace(/\s+/g, " ").trim();
    if (cand.length > 30) continue;
    if (!usedMap[normalizeHeadlineKey_(cand)]) return cand;
  }
  return "";
}

  try {
    const prompt = `
Rewrite this headline to be unique in the same ad while preserving meaning.

${headline}

${adGroup}

Keywords:
${(keywords || []).join(", ")}

Already used headline keys (avoid semantic repeats):
${(usedKeys || []).join(" | ")}

- Max 30 characters

- Return one line only
`;
    const out = callOpenAIText(prompt, { temperature: 0.2 });
    return String(out || "").split(/\r?\n/).map(function(v) { return String(v || "").trim(); }).filter(Boolean)[0] || "";
  } catch (_e) {
    return "";
  }
}

function normalizeOptionBankExtract_(obj) {
  const out = obj || {};
  const norm = function(arr, maxLen) {
    const src = Array.isArray(arr) ? arr : [];
    const rows = [];
    const seen = {};
    for (let i = 0; i < src.length; i++) {
      const item = src[i] || {};
      const text = String(item.text == null ? "" : item.text).replace(/\s+/g, " ").trim();
      if (!text) continue;
      const sliced = text.length > maxLen ? text.slice(0, maxLen).trim() : text;
      const key = sliced.toLowerCase();
      if (!key || seen[key]) continue;
      seen[key] = true;
      rows.push({
        text: sliced,
        theme: String(item.theme == null ? "" : item.theme).trim(),
        angleId: String(item.angleId == null ? "" : item.angleId).toLowerCase().trim(),
        source: String(item.source == null ? "ai" : item.source).trim(),
        score: isFinite(Number(item.score)) ? Number(item.score) : 0
      });
      if (rows.length >= 20) break;
    }
    return rows;
  };
  const headlines = norm(out.headlines, 30);
  const descriptions = norm(out.descriptions, 90);
  while (headlines.length < 20) {
    headlines.push({ text: "", theme: "", angleId: "", source: "ai", score: 0 });
  }
  while (descriptions.length < 20) {
    descriptions.push({ text: "", theme: "", angleId: "", source: "ai", score: 0 });
  }
  return {
    headlines: headlines.slice(0, 20).filter(function(r) { return !!r.text; }),
    descriptions: descriptions.slice(0, 20).filter(function(r) { return !!r.text; })
  };
}

function readBrandOptionBankOptions_(ss) {
  const sheet = getOrCreateBrandOptionBankSheet_(ss);
  const values = sheet.getDataRange().getValues();
  const options = [];
  for (let i = 1; i < values.length; i++) {
    const row = values[i];
    const enabled = toBoolCell_(row[0]);
    if (!enabled) continue;
    const assetType = String(row[1] == null ? "" : row[1]).toLowerCase().trim();
    const optionId = String(row[2] == null ? "" : row[2]).trim();
    const text = String(row[3] == null ? "" : row[3]).trim();
    const theme = String(row[4] == null ? "" : row[4]).trim();
    const angleId = String(row[5] == null ? "" : row[5]).toLowerCase().trim();
    const brandTag = String(row[6] == null ? "" : row[6]).trim();
    const source = String(row[7] == null ? "" : row[7]).trim();
    const score = Number(row[8]);
    if (!optionId || !text) continue;
    if (assetType !== "headline" && assetType !== "description") continue;
    options.push({
      assetType: assetType,
      optionId: optionId,
      text: text,
      theme: theme,
      angleId: angleId,
      brandTag: brandTag,
      source: source,
      score: isFinite(score) ? score : 0
    });
  }
  return options;
}

function readCampaignOptionPicks_(ss) {
  const sheet = getOrCreateCampaignOptionPicksSheet_(ss);
  const values = sheet.getDataRange().getValues();
  const picks = [];
  for (let i = 1; i < values.length; i++) {
    const row = values[i];
    const enabled = toBoolCell_(row[0]);
    if (!enabled) continue;
    const adGroup = String(row[1] == null ? "" : row[1]).trim();
    const assetType = String(row[2] == null ? "" : row[2]).toLowerCase().trim();
    const optionId = String(row[3] == null ? "" : row[3]).trim();
    const targetPosition = Number(row[4]);
    const priority = Number(row[5]);
    const notes = String(row[6] == null ? "" : row[6]).trim();
    if (!adGroup || !optionId) continue;
    if (assetType !== "headline" && assetType !== "description") continue;
    picks.push({
      adGroup: adGroup,
      adGroupKey: normalizeAdGroupKey_(adGroup),
      assetType: assetType,
      optionId: optionId,
      targetPosition: isFinite(targetPosition) ? targetPosition : 0,
      priority: isFinite(priority) ? priority : 9999,
      notes: notes
    });
  }
  return picks.sort(function(a, b) { return a.priority - b.priority; });
}

function applyCampaignOptionPicks() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  ensureOptionBankSheets_();
  const adSheetName = getTextSetting_("AD_SHEET_NAME", "Ad Copy");
  const adSheet = ss.getSheetByName(adSheetName);
  if (!adSheet) {
    SpreadsheetApp.getUi().alert("Ad sheet not found: " + adSheetName);
    return;
  }

  const options = readBrandOptionBankOptions_(ss);
  const picks = readCampaignOptionPicks_(ss);
  const optionsById = {};
  options.forEach(function(o) { optionsById[o.optionId] = o; });
  const fixedAssets = readFixedAssetsConfig_(ss, getTextSetting_("FIXED_ASSETS_SHEET_NAME", DEFAULT_FIXED_ASSETS_SHEET));
  const logSheet = getOrCreateAdGenLogSheet_();

  const data = adSheet.getDataRange().getValues();
  let applied = 0;
  let skipped = 0;

  for (let i = 1; i < data.length; i++) {
    const row = i + 1;
    const adGroup = String(data[i][1] == null ? "" : data[i][1]).trim();
    if (!adGroup) continue;
    const key = normalizeAdGroupKey_(adGroup);
    const rowPicks = picks.filter(function(p) { return p.adGroupKey === key; });
    if (!rowPicks.length) continue;

    const headlines = data[i].slice(3, 18).map(function(v) { return String(v == null ? "" : v).trim(); });
    const descriptions = data[i].slice(18, 22).map(function(v) { return String(v == null ? "" : v).trim(); });

    rowPicks.forEach(function(pick) {
      const option = optionsById[pick.optionId];
      if (!option) {
        skipped++;
        logSheet.appendRow([new Date(), row, adGroup, "OPTION_SKIP", "Missing optionId: " + pick.optionId]);
        return;
      }
      if (option.assetType !== pick.assetType) {
        skipped++;
        logSheet.appendRow([new Date(), row, adGroup, "OPTION_SKIP", "Type mismatch for " + pick.optionId]);
        return;
      }
      const pos = Number(pick.targetPosition);
      if (pick.assetType === "headline") {
        if (!(pos >= 1 && pos <= 15)) {
          skipped++;
          logSheet.appendRow([new Date(), row, adGroup, "OPTION_SKIP", "Invalid headline position for " + pick.optionId]);
          return;
        }
        headlines[pos - 1] = String(option.text || "").slice(0, 30).trim();
        applied++;
      } else {
        if (!(pos >= 1 && pos <= 4)) {
          skipped++;
          logSheet.appendRow([new Date(), row, adGroup, "OPTION_SKIP", "Invalid description position for " + pick.optionId]);
          return;
        }
        descriptions[pos - 1] = String(option.text || "").slice(0, 90).trim();
        applied++;
      }
    });

    applyFixedAssetsByPosition_(headlines, descriptions, fixedAssets, logSheet, row, adGroup);
    const output = headlines.concat(descriptions);
    adSheet.getRange(row, 4, 1, output.length).setValues([output]);
  }

  SpreadsheetApp.getUi().alert("Campaign option picks applied. Updated: " + applied + " | Skipped: " + skipped);
}

function normalizeAdGroupKey_(text) {
  return String(text == null ? "" : text).toLowerCase().trim();
}

function buildOptionId_(assetType, idx) {
  const prefix = assetType === "description" ? "DOPT" : "HOPT";
  const num = String(idx + 1);
  return prefix + (num.length < 2 ? "0" + num : num);
}

function getBlockedTermsByType_(type) {
  const cfg = getPromptConfig_();
  const banned = (cfg.bannedWords || []).map(function(v) { return String(v || "").toLowerCase().trim(); }).filter(Boolean);
  const offShelf = (cfg.offShelfWords || []).map(function(v) { return String(v || "").toLowerCase().trim(); }).filter(Boolean);
  return type === "banned" ? banned : offShelf;
}

function containsAnyTerm_(text, terms) {
  const lower = String(text || "").toLowerCase();
  for (let i = 0; i < terms.length; i++) {
    if (terms[i] && lower.includes(terms[i])) return true;
  }
  return false;
}

function columnToLetter_(column){
  let temp="",letter="";
  while(column>0){
    temp=(column-1)%26;
    letter=String.fromCharCode(temp+65)+letter;
    column=(column-temp-1)/26;
  }
  return letter;
}

function scoreAsset_(text, type, index, adGroup) {

  let score = 100;
  const lower = (text || "").toLowerCase();
  const banned = getBlockedTermsByType_("banned");
  const generic = getBlockedTermsByType_("offShelf");

  if (type === "headline" && text.length > 30) score -= 20;
  if (type === "description" && text.length > 90) score -= 20;
  if (type === "description" && text.length < 60) score -= 15;

  banned.forEach(w => lower.includes(w) && (score -= 25));
  generic.forEach(w => lower.includes(w) && (score -= 20));

  if (text.split(" ").length <= 2) score -= 15;

  if (type === "headline" && index < 3) {
    const words = adGroup.toLowerCase().split(" ");
    const match = words.some(w => lower.includes(w));
    if (!match) score -= 15;
  }

  if (
    !lower.includes("private") &&
    !lower.includes("custom") &&
    !lower.includes("access")
  ) {
    score -= 20;
  }

  return Math.max(score, 0);
}
