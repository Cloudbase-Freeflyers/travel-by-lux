function getSpreadsheetUiSafe_() {
  try {
    return SpreadsheetApp.getUi();
  } catch (_e) {
    return null;
  }
}

function getSetting_(key, fallback) {
  const val = PropertiesService.getScriptProperties().getProperty(key);
  return val === null || val === "" ? fallback : String(val);
}

function getBoolSetting_(key, fallback) {
  return String(getSetting_(key, fallback ? "true" : "false")).toLowerCase() === "true";
}

function setSetting_(key, value) {
  PropertiesService.getScriptProperties().setProperty(key, String(value == null ? "" : value));
}

function htmlEscape_(text) {
  return String(text == null ? "" : text)
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#39;");
}

function injectMissingDynamicSettings_(opts) {
  const options = opts || {};
  const props = PropertiesService.getScriptProperties();
  const keys = Object.keys(DEFAULTS);
  const injected = [];
  for (let i = 0; i < keys.length; i++) {
    const k = keys[i];
    if (props.getProperty(k) === null) {
      props.setProperty(k, DEFAULTS[k]);
      injected.push(k);
    }
  }
  if (!options.silent) {
    const ui = getSpreadsheetUiSafe_();
    if (ui) ui.alert(injected.length ? ("Injected missing settings: " + injected.length) : "No missing settings found.");
  }
  return { injectedCount: injected.length, injectedKeys: injected };
}

function resetDynamicSettings() {
  const keys = Object.keys(DEFAULTS);
  for (let i = 0; i < keys.length; i++) setSetting_(keys[i], DEFAULTS[keys[i]]);
  const ui = getSpreadsheetUiSafe_();
  if (ui) ui.alert("Dynamic settings reset to defaults.");
}

function runOptionalModuleFn_(fnName, moduleFileHint) {
  const fn = this && this[fnName];
  if (typeof fn === "function") return fn();
  const ui = getSpreadsheetUiSafe_();
  if (ui) {
    ui.alert(
      "Function not available: " + fnName + "\n" +
      "Please make sure " + moduleFileHint + " is added and saved in this Apps Script project."
    );
  }
}

function saveSettingsFromSidebar(payload) {
  payload = payload || {};
  if (Object.prototype.hasOwnProperty.call(payload, "openaiApiKey")) {
    const key = String(payload.openaiApiKey || "").trim();
    if (key) setSetting_("OPENAI_API_KEY", key);
  }
  const map = {
    brandName: "BRAND_NAME",
    brandWebsite: "BRAND_WEBSITE",
    positioning1: "POSITIONING_LINE_1",
    positioning2: "POSITIONING_LINE_2",
    customerPromise: "CUSTOMER_PROMISE",
    cta: "CTA",
    ctaMode: "CTA_MODE",
    ctaBrandedList: "CTA_BRANDED_LIST",
    ctaExperienceList: "CTA_EXPERIENCE_LIST",
    angleMinHeadlines: "ANGLE_REQUIRED_MIN_HEADLINES",
    angleMinDescriptions: "ANGLE_REQUIRED_MIN_DESCRIPTIONS",
    bannedWords: "BANNED_WORDS",
    offShelfPhrases: "OFF_SHELF_WORDS",
    customPromptAppend: "CUSTOM_PROMPT_APPEND",
    adTypeTabs: "AD_TYPE_TABS",
    adTypeSheetMap: "AD_TYPE_SHEET_MAP",
    adSheetName: "AD_SHEET_NAME",
    keywordSheetName: "KEYWORD_SHEET_NAME",
    audienceSheetName: "AUDIENCE_SHEET_NAME",
    landingPageSheetName: "LANDING_PAGE_SHEET_NAME"
  };
  const keys = Object.keys(map);
  for (let i = 0; i < keys.length; i++) {
    const p = keys[i];
    if (Object.prototype.hasOwnProperty.call(payload, p)) setSetting_(map[p], payload[p]);
  }
  if (Object.prototype.hasOwnProperty.call(payload, "useKeywords")) setSetting_("USE_KEYWORDS", !!payload.useKeywords);
  if (Object.prototype.hasOwnProperty.call(payload, "useAudiences")) setSetting_("USE_AUDIENCES", !!payload.useAudiences);
  if (Object.prototype.hasOwnProperty.call(payload, "useLandingPageContext")) setSetting_("USE_LANDING_PAGE_CONTEXT", !!payload.useLandingPageContext);
  if (Object.prototype.hasOwnProperty.call(payload, "runOverrideMode")) setSetting_("RUN_OVERRIDE_MODE", !!payload.runOverrideMode);
  return "Settings saved";
}

function normalizeManifestList_(value) {
  if (Array.isArray(value)) return value.map(function(v) { return String(v == null ? "" : v).trim(); }).filter(Boolean).join(",");
  return String(value == null ? "" : value).split(/[\n,]/).map(function(v) { return String(v || "").trim(); }).filter(Boolean).join(",");
}

function extractPromptSettings(payload) {
  const source = payload || {};
  const websiteUrl = String(source.websiteUrl || "").trim();
  const docUrl = String(source.docUrl || "").trim();
  const sourceText = String(source.sourceText || "").trim();
  const apply = !!source.apply;
  if (!websiteUrl && !docUrl && !sourceText) {
    throw new Error("Provide website URL, Google Doc URL, and/or source text.");
  }
  const websiteText = fetchWebsiteTextSafe_(websiteUrl);
  const docRead = getGoogleDocTextFromUrlSafe_(docUrl);
  const docText = docRead.text;
  const prompt = buildExtractionPrompt_(websiteUrl, websiteText, docUrl, docText, sourceText);
  const extracted = callOpenAIJson_(prompt, {
    model: getSetting_("OPENAI_MODEL", DEFAULTS.OPENAI_MODEL),
    temperature: 0.2
  });
  if (apply) applyBrandManifestObject_(extracted);
  const baseMessage = apply ? "Extracted and applied to settings." : "Extracted successfully.";
  const message = docRead.warning ? (baseMessage + " Note: " + docRead.warning) : baseMessage;
  return {
    message: message,
    extracted: extracted,
    manifestJson: JSON.stringify(extracted, null, 2)
  };
}

function getGoogleDocTextFromUrlSafe_(docUrl) {
  const text = String(docUrl || "").trim();
  if (!text) return { text: "", warning: "" };
  const match = text.match(/\/d\/([a-zA-Z0-9_-]+)/);
  if (!match || !match[1]) {
    return {
      text: "",
      warning: "Google Doc URL format was invalid, so extraction ran without Doc content."
    };
  }
  try {
    const doc = DocumentApp.openById(match[1]);
    return { text: String(doc.getBody().getText() || "").slice(0, 24000), warning: "" };
  } catch (_e) {
    return {
      text: "",
      warning: "Google Doc was inaccessible, so extraction ran without Doc content."
    };
  }
}

function fetchWebsiteTextSafe_(url) {
  const target = String(url || "").trim();
  if (!target) return "";
  try {
    const res = UrlFetchApp.fetch(target, { muteHttpExceptions: true, followRedirects: true });
    const html = String(res.getContentText() || "");
    return html
      .replace(/<script[\s\S]*?<\/script>/gi, " ")
      .replace(/<style[\s\S]*?<\/style>/gi, " ")
      .replace(/<[^>]+>/g, " ")
      .replace(/\s+/g, " ")
      .trim()
      .slice(0, 16000);
  } catch (_e) {
    return "";
  }
}

function generateAds() {
  injectMissingDynamicSettings_({ silent: true });
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const adTypes = getAdTypeConfigs_();
  if (!adTypes.length) throw new Error("No ad types configured in AD_TYPE_TABS.");
  const rowLimit = Math.max(1, Number(getSetting_("ROW_LIMIT", DEFAULTS.ROW_LIMIT)) || 5);
  const delayMs = Math.max(0, Number(getSetting_("BATCH_DELAY_MS", DEFAULTS.BATCH_DELAY_MS)) || 0);
  const useKeywords = getBoolSetting_("USE_KEYWORDS", false);
  const useAudiences = getBoolSetting_("USE_AUDIENCES", false);
  const useLandingPageContext = getBoolSetting_("USE_LANDING_PAGE_CONTEXT", false);
  const overrideMode = getBoolSetting_("RUN_OVERRIDE_MODE", false);
  const keywordSheet = useKeywords ? ss.getSheetByName(getSetting_("KEYWORD_SHEET_NAME", DEFAULTS.KEYWORD_SHEET_NAME)) : null;
  const audienceSheet = useAudiences ? ss.getSheetByName(getSetting_("AUDIENCE_SHEET_NAME", DEFAULTS.AUDIENCE_SHEET_NAME)) : null;
  const landingSheet = useLandingPageContext ? ss.getSheetByName(getSetting_("LANDING_PAGE_SHEET_NAME", DEFAULTS.LANDING_PAGE_SHEET_NAME)) : null;
  const keywordData = keywordSheet ? keywordSheet.getDataRange().getValues() : [];
  const audienceData = audienceSheet ? audienceSheet.getDataRange().getValues() : [];
  const landingData = landingSheet ? landingSheet.getDataRange().getValues() : [];
  const landingCache = { byAdGroup: {}, byUrl: {} };
  let processed = 0;
  let stopped = false;

  for (let t = 0; t < adTypes.length; t++) {
    if (isStopRequested_()) { stopped = true; break; }
    if (processed >= rowLimit) break;
    const typeCfg = adTypes[t];
    const sheet = ss.getSheetByName(typeCfg.sheetName);
    if (!sheet) continue;
    const values = sheet.getDataRange().getValues();
    if (values.length < 2) continue;
    const schema = detectAssetColumnsForSheet_(values[0] || []);
    if (schema.adGroupCol < 0) continue;
    const spec = getAssetSpecForAdType_(typeCfg.id, schema);

    for (let r = 1; r < values.length; r++) {
      if (isStopRequested_()) { stopped = true; break; }
      if (processed >= rowLimit) break;
      const row = values[r];
      const adGroup = toText_(row[schema.adGroupCol]);
      if (!adGroup) continue;
      const emptySlots = getEmptyAssetSlots_(row, spec);
      if (!overrideMode && !emptySlots.hasAny) continue;

      const keywords = useKeywords ? getSignalListForAdGroup_(adGroup, keywordData, ["keyword", "keywords"]) : [];
      const audiences = useAudiences ? getSignalListForAdGroup_(adGroup, audienceData, ["audience", "audiences", "signal", "signals"]) : [];
      const landingContext = useLandingPageContext ? getLandingPageContextForAdGroup_(adGroup, landingData, landingCache) : "";
      const prompt = buildPromptForAdType_(typeCfg.id, adGroup, keywords, audiences, landingContext, spec);
      const generated = callOpenAIJson_(prompt, {
        model: getSetting_("OPENAI_MODEL", DEFAULTS.OPENAI_MODEL),
        temperature: Number(getSetting_("OPENAI_TEMPERATURE", DEFAULTS.OPENAI_TEMPERATURE)) || 0.4
      });

      writeAssetsToSheetBySchema_(sheet, r + 1, row, spec, generated, overrideMode);
      processed++;
      if (delayMs > 0) Utilities.sleep(delayMs);
    }
    if (stopped) break;
  }
  SpreadsheetApp.getUi().alert(
    "Generation complete. Rows updated: " + processed +
    (stopped ? "\nStopped early by user request." : "")
  );
}

function getAssetSpecForAdType_(adTypeId, schema) {
  const isPmax = String(adTypeId || "").toLowerCase() === "pmax";
  const headlineMax = 30;
  const descriptionMax = 90;
  const longHeadlineMax = 90;
  const headlineCols = (schema.headlineCols || []).slice(0, isPmax ? 15 : 15);
  const descriptionCols = (schema.descriptionCols || []).slice(0, isPmax ? 5 : 4);
  const longHeadlineCols = isPmax ? (schema.longHeadlineCols || []).slice(0, 5) : [];
  return {

    headlineMax: headlineMax,
    descriptionMax: descriptionMax,
    longHeadlineMax: longHeadlineMax
  };
}

function detectAssetColumnsForSheet_(headerRow) {
  const headers = (headerRow || []).map(function(v) { return toText_(v); });
  const normalize = function(v) { return toText_(v).toLowerCase().replace(/[^a-z0-9]+/g, " ").trim(); };
  const out = { adGroupCol: -1, headlineCols: [], descriptionCols: [], longHeadlineCols: [] };
  const adGroupAliases = { "ad group": true, "asset group": true, "adgroup": true, "assetgroup": true };
  for (let i = 0; i < headers.length; i++) {
    const norm = normalize(headers[i]);
    if (out.adGroupCol < 0 && adGroupAliases[norm]) out.adGroupCol = i;
    let m = norm.match(/^headline\s+(\d+)$/);
    if (m) {
      out.headlineCols.push({ col: i, slot: Number(m[1]) || out.headlineCols.length + 1 });
      continue;
    }
    m = norm.match(/^description\s+(\d+)$/);
    if (m) {
      out.descriptionCols.push({ col: i, slot: Number(m[1]) || out.descriptionCols.length + 1 });
      continue;
    }
    m = norm.match(/^long headline\s+(\d+)$/);
    if (m) {
      out.longHeadlineCols.push({ col: i, slot: Number(m[1]) || out.longHeadlineCols.length + 1 });
    }
  }
  out.headlineCols.sort(function(a, b) { return a.slot - b.slot; });
  out.descriptionCols.sort(function(a, b) { return a.slot - b.slot; });
  out.longHeadlineCols.sort(function(a, b) { return a.slot - b.slot; });
  return out;
}

function getEmptyAssetSlots_(row, spec) {
  const findEmpty = function(cols) {
    const out = [];
    for (let i = 0; i < cols.length; i++) {
      const meta = cols[i];
      const text = toText_(row[meta.col]);
      if (!text) out.push(meta);
    }
    return out;
  };
  const emptyHeadlines = findEmpty(spec.headlineCols || []);
  const emptyDescriptions = findEmpty(spec.descriptionCols || []);
  const emptyLongHeadlines = findEmpty(spec.longHeadlineCols || []);
  return {
    emptyHeadlines: emptyHeadlines,
    emptyDescriptions: emptyDescriptions,
    emptyLongHeadlines: emptyLongHeadlines,
    hasAny: !!(emptyHeadlines.length || emptyDescriptions.length || emptyLongHeadlines.length)
  };
}

function writeAssetsToSheetBySchema_(sheet, rowNumber, currentRow, spec, generated, overrideMode) {
  const headlines = Array.isArray(generated && generated.headlines) ? generated.headlines : [];
  const descriptions = Array.isArray(generated && generated.descriptions) ? generated.descriptions : [];
  const longHeadlines = Array.isArray(generated && generated.longHeadlines) ? generated.longHeadlines : [];

  for (let i = 0; i < spec.headlineCols.length; i++) {
    const meta = spec.headlineCols[i];
    const existing = toText_(currentRow[meta.col]);
    if (existing && !overrideMode) continue;
    const next = enforceMaxLen_(toText_(headlines[i]), spec.headlineMax);
    if (next) sheet.getRange(rowNumber, meta.col + 1).setValue(next);
  }
  for (let i = 0; i < spec.descriptionCols.length; i++) {
    const meta = spec.descriptionCols[i];
    const existing = toText_(currentRow[meta.col]);
    if (existing && !overrideMode) continue;
    const next = enforceMaxLen_(toText_(descriptions[i]), spec.descriptionMax);
    if (next) sheet.getRange(rowNumber, meta.col + 1).setValue(next);
  }
  for (let i = 0; i < spec.longHeadlineCols.length; i++) {
    const meta = spec.longHeadlineCols[i];
    const existing = toText_(currentRow[meta.col]);
    if (existing && !overrideMode) continue;
    const next = enforceMaxLen_(toText_(longHeadlines[i]), spec.longHeadlineMax);
    if (next) sheet.getRange(rowNumber, meta.col + 1).setValue(next);
  }
}

function getSignalListForAdGroup_(adGroup, sheetValues, signalAliases) {
  if (!sheetValues || !sheetValues.length) return [];
  const headers = (sheetValues[0] || []).map(function(v) { return toText_(v); });
  const normalizedHeaders = headers.map(function(v) { return String(v || "").toLowerCase().trim(); });
  const adGroupKey = String(adGroup || "").toLowerCase().trim();
  let adGroupCol = normalizedHeaders.indexOf("ad group");
  if (adGroupCol < 0) adGroupCol = normalizedHeaders.indexOf("asset group");
  let signalCol = -1;
  for (let i = 0; i < signalAliases.length; i++) {
    const idx = normalizedHeaders.indexOf(String(signalAliases[i]).toLowerCase());
    if (idx >= 0) { signalCol = idx; break; }
  }

  const out = [];
  const seen = {};
  if (adGroupCol >= 0 && signalCol >= 0) {
    for (let r = 1; r < sheetValues.length; r++) {
      const rowAdGroup = toText_(sheetValues[r][adGroupCol]).toLowerCase();
      if (rowAdGroup !== adGroupKey) continue;
      const val = toText_(sheetValues[r][signalCol]);
      if (!val) continue;
      const k = val.toLowerCase();
      if (!seen[k]) { seen[k] = true; out.push(val); }
    }
    return out;
  }

  const colByHeader = normalizedHeaders.indexOf(adGroupKey);
  if (colByHeader >= 0) {
    for (let r = 1; r < sheetValues.length; r++) {
      const val = toText_(sheetValues[r][colByHeader]);
      if (!val) continue;
      const k = val.toLowerCase();
      if (!seen[k]) { seen[k] = true; out.push(val); }
    }
  }
  return out;
}

function getLandingPageContextForAdGroup_(adGroup, sheetValues, cache) {
  if (!sheetValues || !sheetValues.length) return "";
  const key = normalizeAdTypeId_(adGroup);
  const memo = cache || { byAdGroup: {}, byUrl: {} };
  if (memo.byAdGroup[key] != null) return memo.byAdGroup[key];
  const url = getLandingPageUrlForAdGroup_(adGroup, sheetValues);
  if (!url) {
    memo.byAdGroup[key] = "";
    return "";
  }
  if (memo.byUrl[url] == null) {
    memo.byUrl[url] = fetchWebsiteTextSafe_(url).slice(0, 1200);
  }
  memo.byAdGroup[key] = memo.byUrl[url] || "";
  return memo.byAdGroup[key];
}

function getLandingPageUrlForAdGroup_(adGroup, sheetValues) {
  if (!sheetValues || !sheetValues.length) return "";
  const headers = (sheetValues[0] || []).map(function(v) { return toText_(v).toLowerCase().trim(); });
  const adGroupKey = toText_(adGroup).toLowerCase().trim();
  let adGroupCol = headers.indexOf("ad group");
  if (adGroupCol < 0) adGroupCol = headers.indexOf("asset group");
  let urlCol = headers.indexOf("landing page");
  if (urlCol < 0) urlCol = headers.indexOf("landing page url");
  if (urlCol < 0) urlCol = headers.indexOf("url");
  if (adGroupCol >= 0 && urlCol >= 0) {
    for (let r = 1; r < sheetValues.length; r++) {
      const rowAdGroup = toText_(sheetValues[r][adGroupCol]).toLowerCase();
      if (rowAdGroup !== adGroupKey) continue;
      const url = toText_(sheetValues[r][urlCol]);
      if (url) return url;
    }
  }
  const colByHeader = headers.indexOf(adGroupKey);
  if (colByHeader >= 0) {
    for (let r = 1; r < sheetValues.length; r++) {
      const val = toText_(sheetValues[r][colByHeader]);
      if (val) return val;
    }
  }
  return "";
}

function callOpenAIJson_(prompt, options) {
  const apiKey = getApiKey_();
  if (!apiKey) throw new Error("OPENAI_API_KEY is missing in script properties.");
  const model = String(options && options.model ? options.model : DEFAULTS.OPENAI_MODEL);
  const temperature = Number(options && options.temperature != null ? options.temperature : DEFAULTS.OPENAI_TEMPERATURE);
  const response = UrlFetchApp.fetch("https://api.openai.com/v1/chat/completions", {
    method: "post",
    contentType: "application/json",
    headers: { Authorization: "Bearer " + apiKey },
    payload: JSON.stringify({
      model: model,
      response_format: { type: "json_object" },
      messages: [{ role: "user", content: prompt }],
      temperature: temperature
    }),
    muteHttpExceptions: true
  });
  const raw = response.getContentText();
  const parsed = JSON.parse(raw);
  const content = parsed && parsed.choices && parsed.choices[0] && parsed.choices[0].message
    ? String(parsed.choices[0].message.content || "")
    : "";
  const cleaned = content.replace(/```json/gi, "").replace(/```/g, "").trim();
  return JSON.parse(cleaned);
}

function getApiKey_() {
  return getSetting_("OPENAI_API_KEY", "");
}

function normalizeAdTypeId_(value) {
  return String(value == null ? "" : value).toLowerCase().trim();
}

function parseCsvList_(value) {
  return String(value == null ? "" : value)
    .split(/[\n,]/)
    .map(function(v) { return String(v || "").trim(); })
    .filter(Boolean);
}

function getEffectiveBlockedTerms_() {
  const fromSettings = parseCsvList_(getSetting_("BANNED_WORDS", DEFAULTS.BANNED_WORDS))
    .map(function(v) { return String(v || "").toLowerCase().trim(); })
    .filter(Boolean);
  const merged = fromSettings.concat(HARD_BLOCKED_TERMS);
  const seen = {};
  const out = [];
  for (let i = 0; i < merged.length; i++) {
    const t = merged[i];
    if (!t || seen[t]) continue;
    seen[t] = true;
    out.push(t);
  }
  return out;
}

function parseKeyValueMap_(value) {
  const out = {};
  const parts = parseCsvList_(value);
  for (let i = 0; i < parts.length; i++) {
    const token = parts[i];
    const idx = token.indexOf(":");
    if (idx <= 0) continue;
    const key = normalizeAdTypeId_(token.slice(0, idx));
    const val = toText_(token.slice(idx + 1));
    if (key && val) out[key] = val;
  }
  return out;
}

function enforceMaxLen_(text, maxLen) {
  const t = normalizeAssetText_(toText_(text));
  if (!t) return "";
  if (t.length <= maxLen) return t;
  const sliced = t.slice(0, maxLen + 1);
  const lastSpace = sliced.lastIndexOf(" ");
  if (lastSpace > Math.floor(maxLen * 0.6)) return sliced.slice(0, lastSpace).trim();
  return t.slice(0, maxLen).trim();
}

function normalizeAssetText_(text) {
  return toText_(text)
    .replace(/([a-z])([A-Z])/g, "$1 $2")
    .replace(/([A-Za-z])\/([A-Za-z])/g, "$1 $2")
    .replace(/\s+/g, " ")
    .trim();
}

function toText_(value) {
  return value == null ? "" : String(value).trim();
}
